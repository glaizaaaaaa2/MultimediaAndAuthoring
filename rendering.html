Solar System

3D rendering converts three-dimensional digital models into imagery using simulated light, materials, and camera perspectives. This project brings the solar system to life through accurate modeling, planetary textures, and orbital mechanics — with an interactive age calculator for every planet.

Tools & Technologies

Production Stack

The following tools and concepts were used in the production of the 3D solar system rendering project.

Modeling
3D Modeling Software
Used to create spherical meshes for each celestial body, set orbital paths, and animate planetary rotations.
Illumination
Lighting & Shading
Point lights simulate the Sun's radiance. Ambient and directional lights create depth and realistic shadow casting.
Textures
Texture Mapping
High-resolution planetary surface maps were UV-mapped onto spherical geometry for photorealism.
Cinematography
Camera & Composition
Animated camera paths orbit the scene to showcase each planet, using depth-of-field to focus attention.
Pipeline
Rendering Engine
A ray-tracing engine converts 3D geometry to 2D pixels with realistic light bounce and shadow.
Post-Production
Video Compositing
Rendered frames composited into a final video with color grading for a cinematic space atmosphere.

ASTEROID_COUNT = 350  
BELT_INNER = 25.0
BELT_OUTER = 29.0
asteroid_list = []
random.seed(42)

for i in range(ASTEROID_COUNT):
    angle = random.uniform(0, 2 * math.pi)
    radius = random.uniform(BELT_INNER, BELT_OUTER)
    z_scatter = random.uniform(-0.4, 0.4)
    rock_scale = random.uniform(0.08, 0.22)
    bpy.ops.mesh.primitive_ico_sphere_add(subdivisions=2, radius=rock_scale)
    rock = bpy.context.active_object
    rock.name = f"Asteroid_{i}"
    rock.data.materials.append(asteroid_mat)
    for vert in rock.data.vertices:
        vert.co += random.uniform(-0.03, 0.03) * vert.co
    planet_collection.objects.link(rock)
    bpy.context.scene.collection.objects.unlink(rock)
    asteroid_list.append({
        "object": rock, "radius": radius, "angle_offset": angle, "z": z_scatter, "orbit_speed": random.uniform(0.15, 0.4)
    })

for frame in range(FRAME_START, FRAME_END + 1):
    bpy.context.scene.frame_set(frame)
    current_planet_positions = {}
    
    for name, data in PLANET_DATA.items():
        obj = created_objects[name]
        rot_angle = math.radians((data["rot_speed"] / 250.0) * frame)
        rev_angle = math.radians((data["orbit_speed"] * 0.4) * frame)
        
        x_pos = data["distance"] * math.cos(rev_angle)
        y_pos = data["distance"] * math.sin(rev_angle)
        
        obj.location = (x_pos, y_pos, 0.0)
        obj.keyframe_insert(data_path="location", frame=frame)
        obj.rotation_mode = 'ZXY'
        obj.rotation_euler = (math.radians(data["tilt"]), 0.0, rot_angle)
        obj.keyframe_insert(data_path="rotation_euler", frame=frame)
        
        current_planet_positions[name] = (x_pos, y_pos)
            
    for moon_data in created_moons:
        m_obj = moon_data["object"]
        p_name = moon_data["parent_name"]
        parent_x, parent_y = current_planet_positions[p_name]
        
        m_angle = math.radians(moon_data["speed"] * frame)
        m_dist = moon_data["distance"]
        
        m_obj.location = (
            parent_x + (m_dist * math.cos(m_angle)),
            parent_y + (m_dist * math.sin(m_angle)),
            0.1
        )
        m_obj.keyframe_insert(data_path="location", frame=frame)
        m_obj.rotation_euler = (0.0, 0.0, math.radians(3.0 * frame))
        m_obj.keyframe_insert(data_path="rotation_euler", frame=frame)
        
    for ast in asteroid_list:
        rock_obj = ast["object"]
        current_angle = ast["angle_offset"] + math.radians(ast["orbit_speed"] * 0.4 * frame)
        rock_obj.location = (ast["radius"] * math.cos(current_angle), ast["radius"] * math.sin(current_angle), ast["z"])
        rock_obj.keyframe_insert(data_path="location", frame=frame)

    if frame <= 900:
        current_showcase_idx = (frame - 1) // showcase_duration

def rigid_hud_text_updater(scene):
    frame = scene.frame_current
    t_obj = bpy.data.objects.get("HUD_Master_Text")
    
    if t_obj:
        if frame <= 900:
            # Force string value updates instantaneously on viewport redraw
            current_idx = (frame - 1) // 100
            if current_idx < len(planet_names):
                t_obj.data.body = planet_names[current_idx].upper()
                t_obj.scale = (1.0, 1.0, 1.0)
        else:
            # Turn off characters layout space entirely during the tour finale
            t_obj.scale = (0.0, 0.0, 0.0)

# Attach script listener directly to Blender background loops queue
