Tools & Technologies

Production Stack

The following tools and concepts were used in the production of the 3D solar system rendering project.

Modeling
3D Modeling Software
Used to create spherical meshes for each celestial body, set orbital paths, and animate planetary rotations.
Illumination
Lighting & Shading
Point lights simulate the Sun's radiance. Ambient and directional lights create depth and realistic shadow casting.
Textures
Texture Mapping
High-resolution planetary surface maps were UV-mapped onto spherical geometry for photorealism.
Cinematography
Camera & Composition
Animated camera paths orbit the scene to showcase each planet, using depth-of-field to focus attention.
Pipeline
Rendering Engine
A ray-tracing engine converts 3D geometry to 2D pixels with realistic light bounce and shadow.
Post-Production
Video Compositing
Rendered frames composited into a final video with color grading for a cinematic space atmosphere.

Python Source Code

Solar System Render — Script

The Python script used to generate the 3D solar system rendering. Uses matplotlib and numpy — no extra installs needed beyond a standard Python environment.

solar_system.py 
# ─────────────────────────────────────────────────────────────────
# Solar System 3D Render
# Author : Glaiza Marie D. Reblinca
# Course : Multimedia & Authoring — BSIT-NS-2A
# Requires: Python 3.8+, numpy, matplotlib
# Run     : python solar_system.py
# ─────────────────────────────────────────────────────────────────

import numpy as np
import matplotlib.pyplot as plt
import matplotlib.animation as animation
from mpl_toolkits.mplot3d import Axes3D
from matplotlib.patches import FancyArrowPatch
import matplotlib.colors as mcolors

# ── Planet data (name, radius, orbit_radius, period, color, tilt) ──
PLANETS = [
    {'name': 'Mercury', 'r': 0.38, 'orbit': 5.0,  'period': 0.241, 'color': '#b5b5b5', 'tilt': 0.03},
    {'name': 'Venus',   'r': 0.95, 'orbit': 7.5,  'period': 0.615, 'color': '#e8cda0', 'tilt': 0.05},
    {'name': 'Earth',   'r': 1.00, 'orbit': 10.0, 'period': 1.000, 'color': '#3a7bd5', 'tilt': 0.08},
    {'name': 'Mars',    'r': 0.53, 'orbit': 13.0, 'period': 1.881, 'color': '#c1440e', 'tilt': 0.04},
    {'name': 'Jupiter', 'r': 2.80, 'orbit': 19.0, 'period': 11.86, 'color': '#c88b3a', 'tilt': 0.02},
    {'name': 'Saturn',  'r': 2.30, 'orbit': 25.0, 'period': 29.46, 'color': '#e4d191', 'tilt': 0.47},
    {'name': 'Uranus',  'r': 1.60, 'orbit': 31.0, 'period': 84.01, 'color': '#7de8e8', 'tilt': 0.12},
    {'name': 'Neptune', 'r': 1.55, 'orbit': 37.0, 'period': 164.8, 'color': '#3f54ba', 'tilt': 0.09},
]

FRAMES    = 240   # total animation frames
FPS       = 30    # frames per second
SUN_R     = 3.0   # sun sphere radius
STAR_COUNT= 500   # background stars


def sphere(cx=0, cy=0, cz=0, r=1, res=20):
    """Return x,y,z arrays for a sphere surface."""
    u = np.linspace(0, 2*np.pi, res)
    v = np.linspace(0, np.pi, res)
    x = cx + r * np.outer(np.cos(u), np.sin(v))
    y = cy + r * np.outer(np.sin(u), np.sin(v))
    z = cz + r * np.outer(np.ones_like(u), np.cos(v))
    return x, y, z


def orbit_ring(radius, tilt=0, pts=120):
    """Return x,y,z for a tilted circular orbit path."""
    theta = np.linspace(0, 2*np.pi, pts)
    x = radius * np.cos(theta)
    y = radius * np.sin(theta)
    z = tilt   * np.sin(theta) * radius * 0.3
    return x, y, z


def planet_pos(planet, frame):
    """Compute (x, y, z) position of a planet at a given frame."""
    angle  = (2 * np.pi * frame) / (FRAMES * planet['period'])
    r      = planet['orbit']
    x = r * np.cos(angle)
    y = r * np.sin(angle)
    z = planet['tilt'] * np.sin(angle) * r * 0.3
    return x, y, z


def build_stars(n=STAR_COUNT):
    """Scatter random stars across a hemisphere."""
    rng   = np.random.default_rng(42)
    theta = rng.uniform(0, 2*np.pi, n)
    phi   = rng.uniform(0, np.pi, n)
    R     = 80
    sx = R * np.sin(phi) * np.cos(theta)
    sy = R * np.sin(phi) * np.sin(theta)
    sz = R * np.cos(phi)
    return sx, sy, sz


# ── Scene setup ──────────────────────────────────────────────────
fig = plt.figure(figsize=(12, 8), facecolor='black')
ax  = fig.add_subplot(111, projection='3d', facecolor='black')
ax.set_axis_off()
ax.set_xlim(-42, 42)
ax.set_ylim(-42, 42)
ax.set_zlim(-12, 12)
ax.view_init(elev=22, azim=-60)

# Background stars
sx, sy, sz = build_stars()
ax.scatter(sx, sy, sz, s=0.6, c='white', alpha=0.55, depthshade=False)

# Sun glow (multiple semi-transparent spheres)
for scale, alpha in [(1.0, 0.9), (1.35, 0.18), (1.75, 0.07)]:
    xs, ys, zs = sphere(r=SUN_R*scale)
    ax.plot_surface(xs, ys, zs, color='#ffcc44', alpha=alpha,
                    linewidth=0, antialiased=True)

# Orbit rings
for p in PLANETS:
    ox, oy, oz = orbit_ring(p['orbit'], p['tilt'])
    ax.plot(ox, oy, oz, color='white', alpha=0.08, linewidth=0.6)

# Saturn rings (flat torus approximation)
saturn = next(p for p in PLANETS if p['name'] == 'Saturn')
ring_theta = np.linspace(0, 2*np.pi, 60)
ring_artists = []
for ring_r in np.linspace(saturn['r']*1.3, saturn['r']*2.1, 6):
    ring_artists.append(ax.plot([], [], [], color='#d4b483', alpha=0.4, lw=1.2)[0])

# Planet sphere objects (will be updated each frame)
planet_surfs = [None] * len(PLANETS)
labels = []
for i, p in enumerate(PLANETS):
    labels.append(ax.text(0, 0, 0, p['name'], fontsize=5.5,
                           color='white', alpha=0.7, ha='center'))

title_txt = ax.text2D(0.5, 0.97, 'Solar System — 3D Render',
                       transform=ax.transAxes, ha='center',
                       fontsize=13, color='white', alpha=0.8',
                       fontfamily='monospace')


def update(frame):
    # Remove old planet surfaces
    global planet_surfs
    for surf in planet_surfs:
        if surf is not None:
            surf.remove()

    for i, p in enumerate(PLANETS):
        px, py, pz = planet_pos(p, frame)
        xs, ys, zs = sphere(px, py, pz, r=p['r'])
        planet_surfs[i] = ax.plot_surface(xs, ys, zs,
                            color=p['color'], alpha=0.95,
                            linewidth=0, antialiased=True)
        labels[i].set_position_3d((px, py, pz + p['r'] + 0.8))

        # Saturn rings — rotate with planet
        if p['name'] == 'Saturn':
            for j, ring_r in enumerate(np.linspace(p['r']*1.3, p['r']*2.1, 6)):
                rx = px + ring_r * np.cos(ring_theta)
                ry = py + ring_r * np.sin(ring_theta)
                rz = np.full_like(ring_theta, pz + 0.2*np.sin(ring_theta))
                ring_artists[j].set_data_3d(rx, ry, rz)

    # Slowly rotate the camera
    ax.view_init(elev=22, azim=-60 + frame * 0.25)
    return planet_surfs + ring_artists + labels


# ── Animate & save ───────────────────────────────────────────────
ani = animation.FuncAnimation(fig, update, frames=FRAMES,
                               interval=1000/FPS, blit=False)

# Save as MP4 (requires ffmpeg) or comment out to just display
writer = animation.FFMpegWriter(fps=FPS, bitrate=2000,
                                 metadata={'title': 'Solar System'})
ani.save('solar_system_render.mp4', writer=writer, dpi=150)
print('Saved solar_system_render.mp4')
